Exercise 1. Fitting the boarding school flu data using likelihood

data(flu)
sir <- function(t, y, pars){
# State variables
S <- y[1]
I <- y[2]
R <- y[3]
# Parameter values
beta <- pars["beta"]
gamma <- pars["gamma"]
# Equations
dS <- - beta*S*I
dI <- beta*S*I - gamma*I
dR <- gamma*I
# Return list of gradients
out <- c(dS, dI, dR)
list(out)
}
# Evaluate the SIR model
loglik <- function(pars,flu){
init <- c(S=762,I=1,R=0)
out <- ode(y=init, times=flu$day, func = sir,
parms = c(beta = pars[1], gamma = pars[2])) %>% as.data.frame()
ll <- sum(dnorm(x=flu$cases,mean=out$I,sd=1,log=TRUE))
return(ll)
}
# Iterate over a wide range values of beta
beta <- seq(from = 0.001,to = 0.005, length.out = 21)
gamma <- 0.47
out_logl <- tibble(NULL)
for(i in 1:length(beta)) {
logl <- as.numeric(unlist(loglik(c(beta[i], gamma), flu)))
out_logl <- bind_rows(out_logl, tibble(beta = beta[i], loglik = logl))
}

Exercise 1c. How does this value relate to the one estimated using the sum of the squared errors? (10 points)

sse <- function(pars, flu) {
  init <- c(S = 762, I = 1, R = 0)
  out <- ode(y = init, times = flu$day, func = sir, parms = c(beta = pars[1], gamma = pars[2])) %>% as.data.frame()
  sse <- sum((flu$cases - out$I)^2)
  return(sse)
}

out_sse <- tibble(NULL)
for (i in 1:length(beta)) {
  curr_sse <- as.numeric(unlist(sse(c(beta[i], gamma), flu)))
  out_sse <- bind_rows(out_sse, tibble(beta = beta[i], sse = curr_sse))
}

min_sse <- min(out_sse$sse)
sse_beta <- out_sse[out_sse$sse == min_sse, "beta"]
